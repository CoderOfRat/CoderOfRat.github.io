<script>
    var mainView = document.querySelector(".content");
    var moveBlock = document.querySelector(".moveBlock");
    var after = document.querySelector(".after");
    mainView.addEventListener("mousemove",move);
    mainView.addEventListener("mouseout",dis);
    function move(e) {
        moveBlock.style.display = "block";
        after.style.display = "block";
        var point = moveBlock.offsetWidth / 2;
        var maxWay = mainView.offsetWidth - moveBlock.offsetWidth -2;
        //移动块的定位left值
        var lenX = e.pageX - mainView.offsetLeft - point;
        //移动块的定位top值
        var lenY = e.pageY - mainView.offsetTop - point;
        if(lenX < 0){
            moveBlock.style.left = 0 + "px";
            moveBlock.style.top = lenY + "px";
        }else if(lenX > maxWay){
            moveBlock.style.left = maxWay + "px";
            after.style.backgroundPositionX = -3 * maxWay + "px";
        } else {
            moveBlock.style.left = lenX + "px";
            after.style.backgroundPositionX = -3 * lenX + "px";
        }
        if(lenY < 0){
            moveBlock.style.top = 0 + "px";
        }else if(lenY > maxWay){
            moveBlock.style.top = maxWay + "px";
            after.style.backgroundPositionY = -3 * maxWay +"px";
        } else if(lenY > mainView){
            after.style.display = "none";
        } else {
            moveBlock.style.top = lenY + "px";
            after.style.backgroundPositionY = -3 * lenY +"px";
        }
    }
    /*移出鼠标，全部隐藏*/
    function dis() {
        moveBlock.style.display = "none";
        after.style.display = "none";
    }
</script>
