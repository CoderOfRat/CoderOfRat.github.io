/*move.js*/
function css(obj, attr) {
    if (obj.currentStyle) {
        return obj.currentStyle[attr];
    } else {
        return getComputedStyle(obj, false)[attr];
    }
}

function move(obj, json, during, fx, callback) {
    clearInterval(obj.timer)
    var starttime = new Date().getTime();
    var d = during || 1000;
    var j = {};
    for (attr in json) {
        j[attr] = {};
        if (attr == 'opacity') {
            j[attr].b = parseFloat(css(obj, 'opacity'));
        } else {
            j[attr].b = parseInt(css(obj, attr));
        }

        j[attr].c = json[attr] - j[attr].b;
    }
    obj.timer = setInterval(function() {
        var currenttime = new Date().getTime();
        var t = currenttime - starttime;
        if (t >= d) {
            t = d;
            clearInterval(obj.timer);
            callback && callback(obj);
        }
        for (attr in json) {
            var v = Tween[fx || 'linear'](t, j[attr].b, j[attr].c, d);
            if (attr == 'opacity') {
                obj.style.opacity = v;
                obj.style.filter = "alpha(opacity=" + (v * 100) + ")";
            }
            obj.style[attr] = v + 'px';
        }


    }, 16);
}


var Tween = {
    linear: function(t, b, c, d) { //匀速
        return c * t / d + b;
    },
    easeIn: function(t, b, c, d) { //加速曲线
        return c * (t /= d) * t + b;
    },
    easeOut: function(t, b, c, d) { //减速曲线
        return -c * (t /= d) * (t - 2) + b;
    },
    easeBoth: function(t, b, c, d) { //加速减速曲线
        if ((t /= d / 2) < 1) {
            return c / 2 * t * t + b;
        }
        return -c / 2 * ((--t) * (t - 2) - 1) + b;
    },
    easeInStrong: function(t, b, c, d) { //加加速曲线
        return c * (t /= d) * t * t * t + b;
    },
    easeOutStrong: function(t, b, c, d) { //减减速曲线
        return -c * ((t = t / d - 1) * t * t * t - 1) + b;
    },
    easeBothStrong: function(t, b, c, d) { //加加速减减速曲线
        if ((t /= d / 2) < 1) {
            return c / 2 * t * t * t * t + b;
        }
        return -c / 2 * ((t -= 2) * t * t * t - 2) + b;
    },
    elasticIn: function(t, b, c, d, a, p) { //正弦衰减曲线（弹动渐入）
        if (t === 0) {
            return b;
        }
        if ((t /= d) == 1) {
            return b + c;
        }
        if (!p) {
            p = d * 0.3;
        }
        if (!a || a < Math.abs(c)) {
            a = c;
            var s = p / 4;
        } else {
            var s = p / (2 * Math.PI) * Math.asin(c / a);
        }
        return -(a * Math.pow(2, 10 * (t -= 1)) * Math.sin((t * d - s) * (2 * Math.PI) / p)) + b;
    },
    elasticOut: function(t, b, c, d, a, p) { //正弦增强曲线（弹动渐出）
        if (t === 0) {
            return b;
        }
        if ((t /= d) == 1) {
            return b + c;
        }
        if (!p) {
            p = d * 0.3;
        }
        if (!a || a < Math.abs(c)) {
            a = c;
            var s = p / 4;
        } else {
            var s = p / (2 * Math.PI) * Math.asin(c / a);
        }
        return a * Math.pow(2, -10 * t) * Math.sin((t * d - s) * (2 * Math.PI) / p) + c + b;
    },
    elasticBoth: function(t, b, c, d, a, p) {
        if (t === 0) {
            return b;
        }
        if ((t /= d / 2) == 2) {
            return b + c;
        }
        if (!p) {
            p = d * (0.3 * 1.5);
        }
        if (!a || a < Math.abs(c)) {
            a = c;
            var s = p / 4;
        } else {
            var s = p / (2 * Math.PI) * Math.asin(c / a);
        }
        if (t < 1) {
            return -0.5 * (a * Math.pow(2, 10 * (t -= 1)) *
                Math.sin((t * d - s) * (2 * Math.PI) / p)) + b;
        }
        return a * Math.pow(2, -10 * (t -= 1)) *
            Math.sin((t * d - s) * (2 * Math.PI) / p) * 0.5 + c + b;
    },
    backIn: function(t, b, c, d, s) { //回退加速（回退渐入）
        if (typeof s == 'undefined') {
            s = 1.70158;
        }
        return c * (t /= d) * t * ((s + 1) * t - s) + b;
    },
    backOut: function(t, b, c, d, s) {
        if (typeof s == 'undefined') {
            s = 3.70158; //回缩的距离
        }
        return c * ((t = t / d - 1) * t * ((s + 1) * t + s) + 1) + b;
    },
    backBoth: function(t, b, c, d, s) {
        if (typeof s == 'undefined') {
            s = 1.70158;
        }
        if ((t /= d / 2) < 1) {
            return c / 2 * (t * t * (((s *= (1.525)) + 1) * t - s)) + b;
        }
        return c / 2 * ((t -= 2) * t * (((s *= (1.525)) + 1) * t + s) + 2) + b;
    },
    bounceIn: function(t, b, c, d) { //弹球减振（弹球渐出）
        return c - Tween['bounceOut'](d - t, 0, c, d) + b;
    },
    bounceOut: function(t, b, c, d) {
        if ((t /= d) < (1 / 2.75)) {
            return c * (7.5625 * t * t) + b;
        } else if (t < (2 / 2.75)) {
            return c * (7.5625 * (t -= (1.5 / 2.75)) * t + 0.75) + b;
        } else if (t < (2.5 / 2.75)) {
            return c * (7.5625 * (t -= (2.25 / 2.75)) * t + 0.9375) + b;
        }
        return c * (7.5625 * (t -= (2.625 / 2.75)) * t + 0.984375) + b;
    },
    bounceBoth: function(t, b, c, d) {
        if (t < d / 2) {
            return Tween['bounceIn'](t * 2, 0, c, d) * 0.5 + b;
        }
        return Tween['bounceOut'](t * 2 - d, 0, c, d) * 0.5 + c * 0.5 + b;
    }
}
/*index.js*/
window.onload = function () {
    /*获取轮播图主视口*/
    var focus = document.querySelector(".focus");
    /*获取左右切换按钮*/
    var btnNext = focus.querySelector(".arrow-r");
    var btnPre = focus.querySelector(".arrow-l");
    /*获取轮播图胶片带 ul*/
    var uls = focus.querySelector("ul");
    /*获取胶片带上的每一幅图片的承载框集合 li*/
    var lis = uls.querySelectorAll("li");
    /*获取小圆点承载框 为后续遍历添加小圆点提供容器*/
    var ols = focus.querySelector(".circle");
    /*新建全局控制开关  主要用来控制动画的启停 但实质上并未停止定时器，只是控制了定时器中的执行函数*/
    var flag = true;
    /*声明并初始化全局 nums*/
    var nums = 0;
    /*以轮播图片数量为遍历条件，向小圆点承载容器中遍历添加小圆点（即向其中添加li元素），
    并且给第一个小圆点添加class的属性为"current"（在CSS里预定义好的当前选中的样式），
    同时，所有添加的li标签，添加额外的对象属性为index:i 然后通过fatherNode.appendChild(sonNode)
    将所有的li添加到ols中*/
    for(let i = 0;i < lis.length; i++){
        let li = document.createElement("li");
        if(i === 0){
            li.className = "current";
        }
        li.setAttribute("index",i);
        ols.appendChild(li);
    }
    /*以下两句需要写在上述遍历完成之后，因为上述遍历的条件是i < lis.length,而先执行下面两句，
    上述遍历就会多一个小圆点，或者也可以通过i < lis.length -1 进行修正，
    克隆第一张图片，追加到轮播图胶片带的尾部，为了无缝轮播做准备*/
    var liEnd = lis[0].cloneNode(true);
    uls.appendChild(liEnd);
    //获取添加完成后的ol下的小圆点的集合 li
    var olis = ols.querySelectorAll("li");
    //遍历上述集合里的li元素对象，并且添加点击事件
    for(let i = 0;i < lis.length; i++) {
        olis[i].addEventListener("click", function () {
            for (let i = 0; i < olis.length; i++) {
                //遍历所有小圆点，清除所有小圆点的class值
                olis[i].className = "";
            }
            //点击哪个小圆点就给哪个小圆点添加class="current"
            this.className = "current";
            //获取当前点击小圆点的index值
            let num = this.getAttribute("index");
            //设置ul的定位left值为index值乘以一个li的身位距离
            uls.style.left = - num * lis[0].offsetWidth + "px";
            //将全局变量的nums设置为当前下标，为了让定时器与点击圆点同步
            nums = this.getAttribute("index");
            console.log(nums);
        });
    }
    //排他思想 清除所有小圆点的class属性值
    function clear() {
        for (let i = 0; i < olis.length; i++) {
            //遍历所有小圆点，清除所有小圆点的class值
            olis[i].className = "";
        }
    }
    //轮播动画实现 -- 定时器 无缝轮播
    function f() {
        if(flag){
            nums++;
            /*当轮播到uls.querySelectorAll("li").length-1（即最后一张的时候）*/
            if(nums === uls.querySelectorAll("li").length -1){
                clear();
                olis[0].className = "current";
                //先是500ms内切换到下一张，然后执行定位到第一张和执行
                move(uls,{"left": - nums * lis[0].offsetWidth},500,"easeOut",function () {
                    nums = 0;
                    uls.style.left = 0;
                    move(uls,{"left": - nums * lis[0].offsetWidth},500,"easeOut");
                });
            } else {
                clear();
                olis[nums].className = "current";
                move(uls,{"left": - nums * lis[0].offsetWidth},500,"easeOut");
            }
            console.log(nums);
        }
        console.log("========");
    }
    setInterval(f,3000);
    //以下通过flag控制定时器内的执行函数，来控制是否轮播
    //鼠标悬停 显示左右箭头
    focus.addEventListener("mouseover",function () {
        btnNext.style.display = "block";
        btnPre.style.display = "block";
        flag = false;
    });
    focus.addEventListener("mouseout",function () {
        btnNext.style.display = "none";
        btnPre.style.display = "none";
        flag = true;
    });
    btnNext.addEventListener("click",function () {
        flag = true;
        f();
        flag = false;
    });
    //实现上一张效果
    btnPre.addEventListener("click",function(){
        flag = true;
        /*实现第一张到最后一张的无缝切换，原理：如果此时为第一张，点击上一张按钮，则立即
        将轮播图胶片带定位到追加的最后一张（这一张和第一张相同，所以不会看出切换效果），
        然后改变小圆点的当前选中为最后一张（不包含追加的那一张），然后执行move函数，切换到
        最后一张（实质上是从追加的那一张切换到最后一张）*/
        if(nums === 0){
            uls.style.left = -(olis.length) * lis[0].offsetWidth + "px";
            nums = olis.length - 1;
            clear();
            olis[nums].className = "current";
            move(uls,{"left": - nums * lis[0].offsetWidth},500,"easeOut");
        }else{
            nums -= 2;
            f();
        }
        flag = false;
    })
};
