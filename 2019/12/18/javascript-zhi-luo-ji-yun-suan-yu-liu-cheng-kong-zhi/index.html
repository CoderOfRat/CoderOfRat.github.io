运算符

运算符（operator）也被称为操作符，是用于实现赋值、比较和执行算数运算等功能的符号。

1. 算数运算符

加 减 乘 除 取余（取模）%

3 % 5  //3 注意！
5 % 3  //2

浮点数float 在数学运算中 会有精度的问题

如console.log(0.1 + 0.2); //控制台输出为0.30000000000000004

注意：我们不能拿着浮点数作为判断依据 0.1 + 0.2 === 0.3 //false

递增和递减运算符（重点）

为了解决变量的自加1或者自减1，如果需要反复给数字变量添加或减去1，可以使用递增（++）和递减（–）运算符来完成。

在JavaScript中，递增和递减运算符，既可以放在变量的前面，也可以放在变量的后面。放在变量的前面时，称为前置递增/递减，放在后面时，称为后置递增/递减。

区别：

放在前面时，变量先加/减1，然后将结果返回给表达式

放在后面时，先将变量的原值返回给表达式，变量再加/减1

共同点：表达式执行完，变量的值都会加/减1

比如：

var a = 5;
var b = 5;
console.log(--a);//4
console.log(a);//4
//即 --a === 5-1 && a = 5-1
console.log(b--);//5
console.log(b);//4
//即 b-- === 5 && b = b-1

比较运算符（关系运算符）

> < == >= <= != === !==

程序里面的等于符号 是==，默认转换数据类型，会把字符串的数据类型转换为数字型（隐式转换）。只判断值是否相等

程序里面的全等===，是代表数据类型、值完全一样

逻辑运算符

逻辑运算符是用来进行布尔值运算的运算符，其返回值为布尔值

  • 逻辑与： && and 两侧都为true 结果才是true,只要有一侧为false，结果就为false

  • 逻辑或： || or 两侧都为false结果才是false,只要有一侧为true，结果就为true

  • 逻辑非： ! not 也叫取反符 !true == false;

逻辑与/或的 短路运算（逻辑中断）（重要）

短路运算的原理：当有多个表达式时,左边的表达式值可以确定结果时,就不再继续运算右边的表达式的值

/*
先进行Boolean强制转换 遇到"",'',NaN,0,null,undefined 转换为false
只要是真，就继续向下运算，遇到假或者执行到头，则短路（中断运算）返回当前值（即找false假）
*/
var c = 123 && 456;//c = 456;
var c = 123 && 456 && 789;//c = 789;
var c = 123 && 0 && 789;//c = 0;
var c = 123 && 456 && "";//c = "";
console.log("0" && 456 && 456 + "Coder"); // "456Coder"
console.log(456 + "Coder" && "" && "0"); // ""
console.log(456 - "Coder" && "" && "0"); // NaN
console.log(456 + "Coder" && "0" && 0); // 0
// 4. 逻辑或短路运算 
/*如果表达式1 结果为真 则返回表达式1 
如果表达式1 结果为假 则返回表达式2
（即找true真）*/
console.log("0" || 456 || 456 + "Coder"); // "0"
console.log(456 + "Coder" || "" || "0"); // "456Coder"
console.log(456 - "Coder" || "" || "0"); // "0"
console.log(456 - "Coder" || "" || 0); // 0

// 逻辑中断很重要 它会影响我们程序运行结果
var num = 0;
console.log(123 || num++);
console.log(num); // 0
//因为结果123 为真，所以 num++就不会运算了，所以还是0

赋值运算符

= += -= \= *= %=

运算符优先级

() > 一元运算符++ -- ! > 数学运算符 > 关系运算符

拓展：表达式和返回值

表达式：是由数字、运算符、变量等，以能求得数值的、有意义的排列方法，所得的组合。

简单理解：由数字、运算符、变量等组成的有意义的或能求得数值的式子。

表达式的特点：表达式最终都会有一个结果，返回给我们，我们称之为返回值

函数（方法）的特点和表达式的特点有些类似

var num = 1 + 1;
/*右边表达式计算完毕，把返回值赋值给num*/

流程控制

在一个程序执行的过程中，各条代码的执行顺序对程序的结果会有直接影响的。很多时候我们要通过控制代码的执行顺序来实现我们要完成的功能。

简单来说：流程控制就是来控制我们代码按照什么结构顺序来执行

流程控制主要有三种结构，分别是顺序结构、分支结构和循环结构

顺序流程控制

顺序流程控制，是依据代码的先后顺序，依次执行，即基本流程控制，没有特定的语法结构。

分支流程控制if语句

由上到下执行代码的过程中，根据不同的条件，执行不同的路径代码（执行代码多选一的过程），从而得到不同的结果

//语法结构伪代码
if(条件表达式) {
    //执行语句
} else {
    //执行语句2
}
//如果if里面的条件表达式结果为真(true) 则执行花括号里面的执行语句，如果if里面的条件表达式结果为假(false) 则执行花括号里面的执行语句2
if() {
   //执行语句1
}else if(){
     //执行语句2   
}else {
    //执行语句3
}

三元表达式

类似于双分支语句 if…else，可以做一些简单的条件选择。由三元运算符组成的式子，称为三元表达式

条件表达式 ? 表达式1 : 表达式2

var num = 9;
var res = num > 10 ? "大于10" : "不大于10";
console.log(res);//不大于10
/*
用户输入0-59之间的数字
如果数字小于10，就在数字前面补一个0，加"0"进行拼接
*/
var res = "";
var val = prompt("请输入小于大于0小于59之间的数字");
res = parseInt(val) < 10 ? 0 + val : val;
alert(res);

分支流程控制switch语句

switch 转换、开关 case 小例子或者选项的意思

/*
条件表达式的返回值和里面的value对应，则执行相应语句，无对应，执行default里的语句
*/
switch(表达式) {
    case value1:
        执行语句1;
        break;
    case value2:
        执行语句2;
        break
    ...
    default:
        执行最后的语句
}
//样例  相匹配的时候 是全等关系 值和类型都要一样 遇到break才跳出循环（重要）
var sw = "匹配项";
switch(sw) {
    case 1:
        console.log("我是匹配项1");
        break;
    case "匹配项":
        console.log("我是匹配项2");
        break;
    default:
        console.log("没找到匹配项");
        break;
}
//结果为 "我是匹配项2"

switch 在结果为确定值的情况下可以使用

if else if 常用在范围判断 更加灵活

循环结构 重要

循环

目的：可以重复执行某些代码（相同的代码或者不同的代码，因为存在计数器，所以可以通过变量控制执行体的执行内容）

for 循环

被重复的代码 称为循环体

/*var i = 0只会执行一次，作为启动变量，然后执行第二句进行判断，然后执行循环体，然后就是i++，然后一直判断执行循环体，第一句就不再执行，直到不满足条件，结束循环*/
for(var i = 0;i < 50; i++){ // i(index)
    alert(i);// 循环体
}

断点操作：F12 –>source–>点击对应文件–>点击需要设置断点处的行号。watch可以监视变量，点击小加号可以添加需要监视的变量。

双重for循环

很多情况下，单层for 循环并不能满足我们的需求，比如我们要打印一个5行5列的图形，或者有规律的变换，可能一个for循环无法满足我们的需求，那么循环嵌套就是我们的优选了。

循环嵌套是指在一个循环语句中再定义一个循环语句的语法结构，比较典型的就是双重for循环结构。

思路：我们可以把里面的循环看作是外层循环的语句，外层循环一次，里面的循环全部执行

五行五列小星星
/*需要输出五次（外层控制执行内部循环五次） 打印五个小心心在一行的语句（内部语句每次执行打印五颗小心心）*/
var str = "";
for (var i = 1; i <= 5; i++) {
    for (var j = 1; j <= 5; j++) {
        str += "♥ ";
    }
    str += "\n";
}
console.log(str);
五行累增小星星
/*需要输出五次（外层控制执行内部循环五次） 打印从一行一个到一行五个小心心的语句（内部语句每次执行打印从1-5颗小心心，通过外层变量控制判断条件）*/
var str = "";
for (var i = 1; i <= 5; i++) {
    for (var j = 1; j <= i; j++) {
        str += "♥ ";
    }
    str += "\n";
}
console.log(str);
五行累减小星星
var str = "";
for (var i = 1; i <= 5; i++) {// 外层循环控制行数
    for (var j = 5; j >= i; j--) {
        str += "♥ ";
    }
    str += "\n";
}
console.log(str);
/*或者*/
var str = "";
for (var i = 1; i <= 5; i++) {// 外层循环控制行数
    for (var j = i; j <= 5; j++) { // 相当于用外层变量控制了起始变量的值
        str += "♥ ";
    }
    str += "\n";
}
console.log(str);
九九乘法表 重要算法
var str = "";
for (var i = 1; i <= 9; i++) { // 外层循环控制行数
    for (var j = 1; j <= i; j++) { // 内层控制列数，以及每列的数量
        var tab = i * j;
        str += j + "*" + i + "=" + tab + " ";
    }
    str += "\n";
}
console.log(str);

while循环

var i = 1;
while(i <= 100){
    //循环体
    i++; // 操作表达式，避免死循环（完成计数器更新）
}
// 可以实现比较复杂的判断，只根据判断是否成立来执行
var message = prompt("你爱我嘛？");
while (message !== "我爱你" && message !== "爱") {
    message = prompt("你爱我嘛？");
}
alert("我也爱你哟~");

do while循环

先执行循环体，再执行while() 至少会执行一次循环体

do {
    //循环体
    i++;
}while(条件表达式)

continue break

continue：跳出此次循环，继续下一次循环，本次循环continue之后的代码不会执行

for(){
    if(跳出循环的条件){
        continue;// 如果执行了这句代码，本次循环alert(i)将不会执行
    }
    alert(i);
}

break：退出并结束循环
